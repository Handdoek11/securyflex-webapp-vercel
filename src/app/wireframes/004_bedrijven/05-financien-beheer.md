```typescript
interface CreditScoring {
  paymentHistory: number;      // 40% weight - payment timeliness
  accountAge: number;          // 25% weight - relationship duration
  invoiceSize: number;         // 20% weight - average invoice value
  industryRisk: number;        // 10% weight - industry-specific risks
  financialStability: number;  // 5% weight - external credit data
}

const calculateCreditRating = (metrics: CreditScoring): CreditRating => {
  const score = (
    metrics.paymentHistory * 0.4 +
    metrics.accountAge * 0.25 +
    metrics.invoiceSize * 0.2 +
    metrics.industryRisk * 0.1 +
    metrics.financialStability * 0.05
  );

  if (score >= 90) return 'A+';
  if (score >= 80) return 'A';
  if (score >= 70) return 'B';
  if (score >= 60) return 'C';
  return 'D';
};
```
